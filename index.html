import org.biojava.nbio.structure.Structure; import org.biojava.nbio.structure.StructureIO; import org.biojava.nbio.structure.io.FileParsingParameters; import java.io.File; import java.io.IOException; public class PDBFetcher { public static void fetchResolution(String pdbId, String saveDirectory) throws IOException { // Retrieve the mmCIF file FileParsingParameters params = new FileParsingParameters(); params.setLoadChemCompInfo(true); params.setAtomCaThreshold(Integer.MAX_VALUE); // This sets the limit for the number of atoms to read in a Structure object. StructureIO.setAtomCachePath(saveDirectory); Structure structure = StructureIO.getStructure(pdbId, params); // Get the resolution from the structure double resolution = structure.getHeader().getResolution(); // Get the structure method String structureMethod = structure.getHeader().getStructureMethod(); // Get crystallization details String crystallizationDetails = structure.getHeader().getCrystallization(); // Get the name of the PDB entry String pdbName = structure.getHeader().getName(); if (resolution != -1) { System.out.println("The Resolution of PDB ID " + pdbId + " (" + pdbName + ") is " + resolution + " Ã…."); System.out.println("Structure method: " + structureMethod + "."); } else { System.out.println("Resolution data not available for PDB ID " + pdbId + "."); System.out.println("Structure method: " + structureMethod + "."); } if (crystallizationDetails != null && !crystallizationDetails.isEmpty()) { System.out.println("Crystallization details: " + crystallizationDetails); } else { System.out.println("Crystallization details not available."); } } public static void main(String[] args) { String pdbId = "1abc"; // Example PDB ID String saveDirectory = "."; // Directory to save the mmCIF file try { fetchResolution(pdbId, saveDirectory); } catch (IOException e) { e.printStackTrace(); } } }
